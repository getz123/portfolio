Developed a GUI-based quiz application using Tkinter, featuring:

  • Multiple pages with Next/Back navigation
  • Radio buttons for MCQ selection
  • Live score tracking
  • Final result display after last question

Technologies used: Python, Tkinter

import tkinter as tk
from tkinter import messagebox

score = 0

questions = [
    ("What is the capital of France?", ["Paris", "London", "Berlin", "Madrid"], "Paris"),
    ("Which language is used for web apps?", ["Python", "HTML", "C++", "Java"], "HTML"),
    ("Who developed Python?", ["Dennis Ritchie", "Guido van Rossum", "Bjarne Stroustrup", "James Gosling"], "Guido van Rossum")
]

def check_answer(selected, correct):
    global score
    if selected == correct:
        score += 1

def quiz_page(index):
    if index >= len(questions):
        show_result()
        return

    for widget in root.winfo_children():
        widget.destroy()

    question, options, answer = questions[index]

    tk.Label(root, text=question, font=("Arial", 16)).pack(pady=20)
    selected_option = tk.StringVar()

    for option in options:
    
        tk.Radiobutton(root, text=option, variable=selected_option, value=option, font=("Arial", 14)).pack(anchor="w")

    def next_question():
        check_answer(selected_option.get(), answer)
        quiz_page(index + 1)

    tk.Button(root, text="Next", command=next_question).pack(pady=20)

def show_result():
    for widget in root.winfo_children():
        widget.destroy()
    tk.Label(root, text=f"Your Score: {score}/{len(questions)}", font=("Arial", 18)).pack(pady=20)

root = tk.Tk()
root.title("Quiz App")
root.geometry("500x400")
quiz_page(0)
root.mainloop()
